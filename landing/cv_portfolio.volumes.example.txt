# Итоговый docker-compose.yml для ~/cv_portfolio
# Добавлено для brieflybot.pro: extra_hosts + volume landing/www
# Certbot уже использует ./certbot/conf и ./certbot/www — менять не нужно

services:
  web:
    build: .
    container_name: cv_portfolio_app
    restart: unless-stopped
    volumes:
      - ./static:/app/static
      - ./templates:/app/templates
    networks:
      - cv_network
    environment:
      - PYTHONUNBUFFERED=1
    deploy:
      resources:
        limits:
          memory: 400M
        reservations:
          memory: 256M
    depends_on:
      - nginx
    healthcheck:
      test: ["CMD", "python", "-c", "import urllib.request; urllib.request.urlopen('http://localhost:8000/')"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  nginx:
    image: nginx:alpine
    container_name: cv_portfolio_nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    extra_hosts:
      - "host.docker.internal:host-gateway"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/conf.d:/etc/nginx/conf.d:ro
      - ./static:/app/static:ro
      - ./certbot/conf:/etc/letsencrypt:ro
      - ./certbot/www:/var/www/certbot:ro
      - /root/AI-Digest-Bot/landing/www:/var/www/brieflybot.pro:ro
    networks:
      - cv_network
    deploy:
      resources:
        limits:
          memory: 128M
        reservations:
          memory: 64M

networks:
  cv_network:
    driver: bridge
